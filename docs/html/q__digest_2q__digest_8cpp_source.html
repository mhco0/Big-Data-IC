93  void q_digest::private_compress(int cur_node, int debt, int left_range, int right_range){
94  if (cur_node == -1) return;
95  if (tree[cur_node].weight == 0) return;
96 
97  if(left_range == right_range){
98  tree[cur_node].weight -= debt;
99  } else {
100 
101  int all_sub_weight = this->subtree_weight_from(cur_node);
102 
103  if((all_sub_weight - debt) > capacity){
104  int middle = (left_range + right_range) / 2;
105 
106  debt += (capacity - tree[cur_node].weight);
107  tree[cur_node].weight = capacity;
108 
109  //pass the sub-tree weight for the left and the right child
110 
111  int left_sub_weight = this->subtree_weight_from(tree[cur_node].left_child);
112 
113  if(tree[cur_node].left_child != -1){
114  this->private_compress(tree[cur_node].left_child, min(debt, left_sub_weight), left_range, middle);
115  this->private_compress(tree[cur_node].left_child + 1, max(debt - left_sub_weight, 0), middle + 1, right_range);
116  }
117  }else{
118  tree[cur_node].weight = all_sub_weight - debt;
119

121  this->set_subtree_weight_to_zero(tree[cur_node].left_child);
