183  uniform_int_distribution<ll> who_pick(0LL,1LL);
184  uniform_int_distribution<ll> how_many_give(0LL, 1LL);
185  vector<ll> stream(vector_size,0);
186  vector<pair<ll,ll>> commands_to_stream;
187 
188  for(int i=0;i<number_of_add_operations;i++){
189  ll random_element = who_pick(generator);
190 
191  ll delta_to_random_element = how_many_give(generator);
192 
193  stream[random_element] += delta_to_random_element;
194 
195  commands_to_stream.push_back({random_element,delta_to_random_element});
196  }
197 
198  for(int i=0;i<number_of_sub_operations;i++){
199  ll random_element = who_pick(generator);
200  ll delta_to_random_element = max(how_many_give(generator),stream[random_element]);
201 
202  stream[random_element] -= delta_to_random_element;
203 
204  commands_to_stream.push_back({random_element,-delta_to_random_element});
205  }
206 
207  return {stream, commands_to_stream};
208 }
