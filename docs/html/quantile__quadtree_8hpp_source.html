200  void change_box(aabb<int>& cur_box, short direction){
201  int center_x = (cur_box.bounds().second.x() + cur_box.bounds().first.x()) / 2;
202  int center_y = (cur_box.bounds().second.y() + cur_box.bounds().first.y()) / 2;
